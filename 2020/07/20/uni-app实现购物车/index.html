uni-app实现带商店的购物车

开发工具：HBuilder X 使用框架：uni-app colorUI Vue

功能描述（页面布局从上往下描述）

  • 1.删除选中的商品功能
  • 2.布局，两个盒子，显示商家信息，具体商品图片，价格，数量加减
  • 3.全选与反选，合计总价，前往结算
  • 4.scroll与防抖

具体代码如下

                    <view class="flex align-center" style="margin-bottom: 20upx;border-bottom: 1px solid #DDDDDD;">

                    <view class="flex align-center">
                        <checkbox-group style="margin-right: 10upx;">
                            <checkbox @tap="changeChecked(index,indexs)" style="transform: scale(0.7);" class="round sx-check" :class="row.checked==true?'checked':''" :checked="row.checked" ></checkbox>

            return {  
                // 商品列表，真实开发环境下，调接口来填充
                lists: [],

                checked: true,
                scrollTop: 0,
                isShowTop: false,
                old: {
                    scrollTop: 0
                },
                checkedList: []
            }
        },
        //onload时，加载数据
        onLoad(e) {
            console.log(this.lists);
            this.getList();
        },
        computed:{
            //在计算属性中，进行总价的计算
            sum(){
                var ots = this.lists;
                // console.log(ots);
                var he = 0;
                this.lists.forEach(item => {
                    item.shoplist.forEach(item1 => {
                        if(item1.checked == true){
                            he += item1.num * item1.price;
                        }
                    })                    
                });
                return he.toFixed(2);
            }
        },
        components: {
            //声明组件
            uniNumberBox
            }, 
        methods: {
            //选中商店
            checkedShop(index){
                this.lists[index].checked = this.lists[index].checked?false:true;
                for (let i = 0; i < this.lists[index].shoplist.length; i++) {
                                    this.lists[index].shoplist[i].checked = this.lists[index].checked;
                                }
                                let a = 0;
                                for (var i = 0; i < this.lists.length; i++) {
                                    if(this.lists[i].checked){
                                        a++; 
                                    }else{
                                        this.isALLchecked  = false
                                    }
                                }
                                if(a == this.lists.length){
                                    this.isALLchecked  = true
                                }
            },
            //单选改变选中状态
            changeChecked(index,indexs){
                // this.lists[index].checked = this.lists[index].checked?false:true;
                // var selectd = this.lists.filter(item => {return !item.checked});
                // selectd.length==0?this.checked=true:this.checked=false;
                this.checkedList = [];
                this.lists[index].shoplist[indexs].checked = this.lists[index].shoplist[indexs].checked?false:true;
                var a = 0;
                for (var i = 0; i < this.lists[index].shoplist.length; i++) {
                    if(this.lists[index].shoplist[i].checked){
                        a++
                    }
                };
                if (a==this.lists[index].shoplist.length){
                                     this.lists[index].checked = true;
                                } else {
                                    this.lists[index].checked = false;
                                }

                                var b = 0;
                                for(var j = 0 ; j < this.lists.length ; j++){
                                    if(this.lists[j].checked){
                                        b++;
                                    }
                                }
                                if (b==this.lists.length){
                                    this.isALLchecked = true
                                } else {
                                    this.isALLchecked  = false;
                                }
            },
            changeMath(val){
                if(val == 0){

                }
            },
            //全选
            changeAll(e){
                let len = this.lists.length;
                    let arr = [];
                    for(let i=0;i<len;i++){  
                        this.lists[i].checked = !this.isALLchecked 
                                 for(let j = 0 ; j < this.lists[i].shoplist.length;j++){ 
                                     this.lists[i].shoplist[j].checked = this.isALLchecked? false : true;
                                     arr.push(this.lists[i].shoplist[j].id); 
                                 } 
                    }
                    this.checkedList = this.isALLchecked?[]:arr;
                    this.isALLchecked = this.isALLchecked||this.lists.length==0?false : true;
            },
            // 删除选中的商品
            deleteGoods() {
                var that = this;
                uni.showModal({
                    confirmText:"确定",
                    content: "您确定要把商品移除购物车吗",
                    title: "删除提示",
                    success(res) {
                        let len = that.lists.length;
                        var a =[];
                        for(let i = 0 ; i < len;i++){                         
                                if(that.lists[i].checked){    
                                 a.push(i);
                                } 
                                // 此处需要注意，我们在多选删除的时候，是按照序号从后往前删除的，按顺序删除不能达到效果
                                // 此外也不能使用forEach，因为该方法开始执行时会自动缓存数组的长度，进行删除数组元素时会出现意外
                                for(let j = 0 ; j < that.lists[i].shoplist.length; j++){
                                        if (that.lists[i].shoplist[j].checked){
                                            that.lists[i].shoplist.splice(j,1);
                                            j--
                                    }
                                } 
                        }
                        for(let k = 0 ; k < a.length; k++){
                            that.lists.splice(a[k],1);          
                        }
                        }
                    }
                })
            },
            //监控滚动
            Scroll(e){

                this.old.scrollTop = e.detail.scrollTop
                this.old.scrollTop = e.detail.scrollTop
                if (this.old.scrollTop > 100){
                    this.isShowTop = true
                }else {
                    this.isShowTop = false
                }

            },
            //获取数据
            getList(){
